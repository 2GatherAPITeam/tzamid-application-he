בִּקְצֵה כְּפָר אֶחָד חַיָה פַּעַם אִשָׁה זְקֵנָה וַחֲכָמָה. הִיא הָיְתָה מַמְצִיאָה גְדוֹלָה וְכָל הַכְּפָר נֶהֱנָה וְשִׂגְשֵׂג בִּזְכוּת הַמְצָאוֹתֶיהָ.
הִיא, לְמָשָׁל, הִמְצִיאָה קֶשֶׁת וְחֵץ שֶׁלְעוֹלָם לֹא מַחְמִיצִים אֶת הַמַטָרָה. צַיָדֵי הַכְּפָר עָשׂוּ שִׁמוּשׁ רַב בַּהַמְצָאָה הַזֹאת. הִיא הִמְצִיאָה גַם מַחֲרֵשָׁה מְיֻחֶדֶת שֶׁקִצְרָה מְאֹד אֶת זְמַן הַחֲרִישָׁה בַּשָׂדֶה. הִיא הִמְצִיאָה מְזוֹן עוֹפוֹת שֶׁהִכְפִּיל אֶת מִסְפַּר הַבֵּיצִים שֶׁהִטִילוּ הַתַרְנְגוֹלוֹת, זְרוֹעוֹת אוֹטוֹמָטִיוֹת שֶׁאָסְפוּ אֶת הַבֵּיצִים הָרַבּוֹת, תַנוּרִים שֶׁאָפוּ עוּגוֹת נֶהֱדָרוֹת מֵהַבֵּיצִים וַאֲפִלוּ מַכְשִׁיר מְסֻבָּךְ שֶׁנִקָה אַחֲרֵי הָאֲרוּחוֹת.

יוֹם אֶחָד יָשְׁבָה הַמַמְצִיאָה הַזְקֵנָה בַּמַעְבָּדָה שֶׁלָה. הִיא עִרְבְּבָה מְעַט מִצֶמַח אֶחָד וּמְעַט מִצֶמַח אַחֵר, הוֹסִיפָה וְעִרְבְּבָה, סִנְנָה וְהֵרִיחָה, וְהִנֵה הִיא הִמְצִיאָה תַבְלִין מְיֻחָד. זֶה הָיָה תַבְלִין שֶׁכְּמוֹתוֹ לֹא נִרְאָה בָּעוֹלָם כֻּלוֹ. תַבְלִין שֶׁאִם מוֹסִיפִים אוֹתוֹ לְכָל תַבְשִׁיל שֶׁהוּא, הוּא הוֹפֵךְ אֶת טַעֲמוֹ לַטַעַם הַמְשֻׁבָּח בְּיוֹתֵר שֶׁיֵשׁ. הַמַמְצִיאָה שָׂמְחָה מְאֹד בְּהַמְצָאָתָה הַגְדוֹלָה וְהִצִיעָה אֶת הַתַבְלִין הַמְיֻחָד לְאַנְשֵׁי הַכְּפָר.

הָרִאשׁוֹנָה שֶׁנִסְתָה אוֹתוֹ הָיְתָה הָאוֹפָה מִמוֹרַד הָרְחוֹב. הִיא הוֹסִיפָה מְעַט מֵהַתַבְלִין לַלְחָמִים שֶׁאָפְתָה. בַּבֹּקֶר קָנוּ הָאֲנָשִׁים בַּכְּפָר לֶחֶם וְלַחְמָנִיוֹת בַּמַאֲפִיָה כְּמוֹ תָמִיד, אַךְ מִיָד אַחֲרֵי שֶׁנָגְסוּ בַּמַאֲפִים חָזְרוּ וּבִקְשׁוּ מֵהָאוֹפָה לְסַפֵּר לָהֶם אֶת סוֹד הַטַעַם הָאֱלֹהִי. הָאוֹפָה הָיְתָה אִשָׁה טוֹבָה וְלָכֵן סִפְּרָה לְכֻלָם עַל הַתַבְלִין הַמְיֻחָד שֶׁמָכְרָה לָה הַמַמְצִיאָה הַגְדוֹלָה שֶׁבִּקְצֵה הַכְּפָר.
